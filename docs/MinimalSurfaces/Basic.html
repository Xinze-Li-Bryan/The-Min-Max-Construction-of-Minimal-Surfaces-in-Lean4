    class MinimalSurfaces.ClosedRiemannian3Manifold (M : Type u_1) extends MetricSpace M, CompactSpace M :
    Type u_1

    3维闭黎曼流形（简化版本）

    • dist : M → M → ℝ
    • dist_self (x : M) : dist x x = 0
    • dist_comm (x y : M) : dist x y = dist y x
    • dist_triangle (x y z : M) : dist x z ≤ dist x y + dist y z
    • edist : M → M → ENNReal
    • edist_dist (x y : M) : edist x y = ENNReal.ofReal (dist x y)
    • toUniformSpace : UniformSpace M
    • uniformity_dist : uniformity M = ⨅ (ε : ℝ), ⨅ (_ : ε > 0), Filter.principal {p : M × M | dist p.1 p.2 < ε}
    • toBornology : Bornology M
    • cobounded_sets : (Bornology.cobounded M).sets = {s : Set M | ∃ (C : ℝ), ∀ x ∈ sᶜ, ∀ y ∈ sᶜ, dist x y ≤ C}
    • eq_of_dist_eq_zero {x y : M} : dist x y = 0 → x = y
    • isCompact_univ : IsCompact Set.univ
    • dim_three : ℕ
    • smooth_structure : True
    • riemannian_metric : True
